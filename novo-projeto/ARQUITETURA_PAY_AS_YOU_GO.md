### 1. Dashboard de Assinaturas

```sql
-- MÃ©tricas gerais
SELECT 
    COUNT(*) FILTER (WHERE status = 'ATIVA') as assinaturas_ativas,
    COUNT(*) FILTER (WHERE status = 'TRIAL') as em_trial,
    COUNT(*) FILTER (WHERE status = 'CANCELADA') as canceladas,

```

### 2. Assinaturas PrÃ³ximas de Expirar

```sql
-- Assinaturas que expiram nos prÃ³ximos 7 dias
SELECT 
    a.id,
    a.usuario_id,
    p.nome as plano,
    a.expira_em,
    EXTRACT(DAY FROM (a.expira_em - CURRENT_TIMESTAMP)) as dias_restantes
FROM assinaturas a
JOIN planos_pagamento p ON a.plano_id = p.id
WHERE a.status IN ('ATIVA', 'TRIAL')
  AND a.expira_em BETWEEN CURRENT_TIMESTAMP AND CURRENT_TIMESTAMP + INTERVAL '7 days'
ORDER BY a.expira_em;
```

### 3. Receita por Plano

```sql
-- Receita mensal por plano
SELECT 
    p.nome,
    COUNT(a.id) as quantidade_assinaturas,
    p.preco_mensal,
    COUNT(a.id) * p.preco_mensal as receita_mensal_total
FROM planos_pagamento p
LEFT JOIN assinaturas a ON p.id = a.plano_id AND a.status = 'ATIVA'
GROUP BY p.id, p.nome, p.preco_mensal
ORDER BY receita_mensal_total DESC;
```
